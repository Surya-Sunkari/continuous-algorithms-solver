Let $\{\mathcal{T}_{\mathbf{x}}\}_{\mathbf{x} \in \mathbb{R}^d}$ be transition distributions corresponding to a random walk on $\mathbb{R}^d$ with stationary distribution $\pi^\star$, and let $\pi$ be a density on $\mathbb{R}^d$ which is $\beta$-warm with respect to $\pi^\star$.

\begin{enumerate}[(i)]
  \item Prove that $D_{\mathrm{TV}}(\mathcal{T}\pi, \pi^\star) \leq D_{\mathrm{TV}}(\pi, \pi^\star)$.
  \item Prove that $\mathcal{T}\pi$ is $\beta$-warm with respect to $\pi^\star$.
\end{enumerate}
