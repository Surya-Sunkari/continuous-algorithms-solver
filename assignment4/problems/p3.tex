\section*{Problem 3}

Let $G = (V, E, \mathbf{w})$ be an undirected graph with Laplacian matrix $\mathbf{L}_G$ (Definition 5, Part II).

\begin{enumerate}[(i)]
    \item What is the kernel of $\mathbf{L}_G$?
    \item Give a combinatorial interpretation of the value of $\mathbf{x}^\top \mathbf{L}_G \mathbf{x}$, when $\mathbf{x} \in \{-\frac{1}{2}, \frac{1}{2}\}^V$.
    \item Prove that there exists a graph $H = (V, E', \mathbf{w}')$ on the same vertex set $V$, with $E' \subseteq E$, $\mathbf{w}' \in \mathbb{R}^{E'}_{>0}$, and Laplacian matrix $\mathbf{L}_H$, such that $|E'| = O(|V| \log |V|)$, and\footnote{Section 3, Part IX may be helpful for this.}
    \[
        0.9 \mathbf{L}_G \preceq \mathbf{L}_H \preceq 1.1 \mathbf{L}_G.
    \]
\end{enumerate}
