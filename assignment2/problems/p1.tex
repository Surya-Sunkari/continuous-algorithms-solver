\section*{Problem 1}

Let $f : \mathbb{R}^d \to \mathbb{R}$ be a differentiable, convex function, and for all $\lambda > 0$, let
\[
\mathbf{x}_\lambda^\star := \operatorname{argmin}_{\mathbf{x} \in \mathbb{R}^d} f(\mathbf{x}) + \frac{\lambda}{2} \|\mathbf{x}\|_2^2.
\]

\begin{enumerate}[(i)]
\item Prove that for any $\lambda' > \lambda > 0$, we have $\|\mathbf{x}_\lambda^\star\|_2 \geq \|\mathbf{x}_{\lambda'}^\star\|_2$.
\item Suppose $\|\nabla f(\mathbf{0}_d)\|_2 \leq L$ and let $R > 0$. Give a $\lambda_R$ such that for any $\lambda \geq \lambda_R$, $\|\mathbf{x}_\lambda^\star\|_2 \leq R$.
\end{enumerate}
