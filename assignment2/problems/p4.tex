\section*{Problem 4}

\begin{enumerate}[(i)]
    \item Prove that the following $2d \times 2d$ matrix is positive semidefinite, for any $a \in \mathbb{R}$ and $\mathbf{M} \in \mathbb{S}_{\succeq \mathbf{0}}^{d \times d}$:
    \[
    \begin{pmatrix} a^2 \mathbf{M} & a\mathbf{M} \\ a\mathbf{M} & \mathbf{M} \end{pmatrix}.
    \]

    \item Prove that for any $\mathbf{a} \in \mathbb{R}^n$, and $\{\mathbf{M}_i\}_{i \in [n]} \subset \mathbb{S}_{\succeq \mathbf{0}}^{d \times d}$ satisfying $\sum_{i \in [n]} \mathbf{M}_i \preceq \mathbf{I}_d$, we have\footnote{This can be viewed as a matrix extension of the Cauchy-Schwarz inequality.}
    \[
    \left( \sum_{i \in [n]} a_i \mathbf{M}_i \right)^2 \preceq \sum_{i \in [n]} a_i^2 \mathbf{M}_i.
    \]
\end{enumerate}
