\section*{Problem 2}

Let $\mathbf{M} \in \mathbb{S}^{d \times d}_{\succ 0}$, and let $\|\cdot\|$ be a norm on $\mathbb{R}^d$.

\begin{enumerate}[(i)]
    \item Prove that if $\mathbf{M}[\mathbf{v}, \mathbf{v}] \leq \|\mathbf{v}\|^2$ for all $\mathbf{v} \in \mathbb{R}^d$, then $\|\mathbf{M}\mathbf{v}\|_* \leq \|\mathbf{v}\|$ for all $\mathbf{v} \in \mathbb{R}^d$.\footnote{This justifies that for general norms, the second claim in Lemma 14, Part II is true. Note that the last part of the proof of Lemma 6, Part II implicitly used this claim in the $\ell_2$ case.}

    \item Prove that $\mathbf{M}[\mathbf{v}, \mathbf{v}] \leq \|\mathbf{v}\|^2$ for all $\mathbf{v} \in \mathbb{R}^d$ iff $\mathbf{M}^{-1}[\mathbf{v}, \mathbf{v}] \geq \|\mathbf{v}\|_*^2$ for all $\mathbf{v} \in \mathbb{R}^d$.\footnote{Combined with $\nabla^2 f^*(\nabla f(\mathbf{x})) = (\nabla^2 f(\mathbf{x}))^{-1}$, shown in class when $f, f^*$ are twice-differentiable, this gives a simple proof of smoothness-strong convexity duality for twice-differentiable convex functions in general norms.}
\end{enumerate}
