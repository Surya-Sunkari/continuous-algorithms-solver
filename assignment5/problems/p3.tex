\begin{enumerate}[(i)]
    \item Let $\pi \in \mathcal{P}(\mathbb{R}^d)$, and let $f : \mathbb{R}^d \to \mathbb{R}$ satisfy $\int f(\mathbf{x})^2 \pi(\mathbf{x}) \mathrm{d}\mathbf{x} < \infty$. Prove that
    \[
        \mathrm{Var}_\pi[f] = \inf_{m \in \mathbb{R}} \mathbb{E}_\pi\left[(f(\mathbf{x}) - m)^2\right].
    \]

    \item Let $\mu, \pi \in \mathcal{P}(\mathbb{R}^d)$ have $c \leq \frac{\mu(\mathbf{x})}{\pi(\mathbf{x})} \leq C$ for all $\mathbf{x} \in \mathbb{R}^d$, and let $\pi$ satisfy a Poincar\'e inequality with constant $C_{\mathrm{PI}}$. Prove that $\mu$ satisfies a Poincar\'e inequality with constant $\frac{C}{c} \cdot C_{\mathrm{PI}}$.
\end{enumerate}
