Let $\pi = \mathcal{N}(\boldsymbol{\mu}, \mathbf{A})$, $\pi' = \mathcal{N}(\boldsymbol{\nu}, \mathbf{B})$ be multivariate normal densities on $\mathbb{R}^d$, with respective means and full-rank covariance matrices $(\boldsymbol{\mu}, \mathbf{A}) \in \mathbb{R}^d \times \mathbb{S}_{>\mathbf{0}}^{d \times d}$ and $(\boldsymbol{\nu}, \mathbf{B}) \in \mathbb{R}^d \times \mathbb{S}_{>\mathbf{0}}^{d \times d}$.

\begin{enumerate}[(i)]
    \item Brenier's theorem (see Proposition 4, Part XVI) states there is a unique optimal transport map $\nabla \varphi$ sending $\mathbf{x} \sim \pi$ to $\mathbf{y} = \nabla \varphi(\mathbf{x}) \sim \pi'$, for convex $\varphi : \mathbb{R}^d \to \mathbb{R}$, such that $(\mathbf{x}, \nabla \varphi(\mathbf{x}))$ for $\mathbf{x} \sim \pi$ is the coupling realizing $W_2^2(\pi, \pi')$. What is $\varphi$ in the above setting?\footnote{The identity $\mathbf{A}^{-\frac{1}{2}}(\mathbf{A}^{\frac{1}{2}} \mathbf{B} \mathbf{A}^{\frac{1}{2}})^{\frac{1}{2}} \mathbf{A}^{-\frac{1}{2}} = \mathbf{A}^{-1}(\mathbf{A}\mathbf{B})^{\frac{1}{2}}$ may be helpful.}

    \item Compute $W_2^2(\pi, \pi')$ in the above setting.
\end{enumerate}
