\section*{Problem 4}

For $\mathbf{K} \in \mathbb{S}^{d \times d}_{\succ \mathbf{0}}$, define the quantity
\[
\kappa^\star_{\mathrm{diag}}(\mathbf{K}) := \min_{\substack{\mathbf{W} \in \mathbb{S}^{d \times d}_{\succ \mathbf{0}} \\ \mathbf{W} \text{ is diagonal}}} \frac{\boldsymbol{\lambda}_1\left(\mathbf{W}^{-\frac{1}{2}} \mathbf{K} \mathbf{W}^{-\frac{1}{2}}\right)}{\boldsymbol{\lambda}_d\left(\mathbf{W}^{-\frac{1}{2}} \mathbf{K} \mathbf{W}^{-\frac{1}{2}}\right)}.
\]

\begin{enumerate}[(i)]
\item Prove that $\kappa^\star_{\mathrm{diag}}(\mathbf{K}) \le \kappa$ iff there is a diagonal matrix $\mathbf{W} \in \mathbb{S}^{d \times d}_{\succ \mathbf{0}}$ such that
\[
\mathbf{W} \preceq \mathbf{K} \preceq \kappa \mathbf{W}.
\]

\item Prove that if $\mathbf{D} \in \mathbb{S}^{d \times d}_{\succ \mathbf{0}}$ is a diagonal matrix,
\[
\kappa^\star_{\mathrm{diag}}(\mathbf{K} + \mathbf{D}) \le \kappa^\star_{\mathrm{diag}}(\mathbf{K}).
\]
\end{enumerate}
